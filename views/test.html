  //assert.equal('41', fkey);

Poster

Create a poster

Create a poster with a prepend

accepts a room and a prepend

Set a poster to silent

accepts a value of true
var msgposter = new poster.MessagePoster(room);
var val = msgposter.silent(true);
assert.ok(val);

Get a posters value of silent

should return false
var msgposter = new poster.MessagePoster(room);
var val = msgposter.silent();
assert.equal(false, val);

Toggle a posters value of silent

should toggle
var msgposter = new poster.MessagePoster(room);
var toggle = msgposter.silent();
var val = msgposter.silent(!toggle);
assert.equal(!toggle, val);
